// ── RATE LIMITING ─────────────────────────────────────────────────────────────
const _rateStore = new Map();
const RateLimit = {
  check(key, max, windowMs) {
    const now=Date.now(), start=now-windowMs;
    if(!_rateStore.has(key)) _rateStore.set(key,[]);
    const hits=_rateStore.get(key).filter(ts=>ts>start);
    _rateStore.set(key,hits);
    return {allowed:hits.length<max,count:hits.length,remaining:Math.max(0,max-hits.length)};
  },
  record(key) { if(!_rateStore.has(key))_rateStore.set(key,[]); _rateStore.get(key).push(Date.now()); },
};

function checkLimit(user) {

  const limit=user.isPremium?LIMITS.premium:LIMITS.free;
  const count=getDailyCount(user.id);

  return {allowed:count<limit&&burst.allowed,count,limit,isPrem:user.isPremium,burstBlocked:!burst.allowed};
}

// ── PASSWORD HASH ─────────────────────────────────────────────────────────────
async function hashPwd(pw) {
  try {
    const buf=await crypto.subtle.digest("SHA-256",new TextEncoder().encode(pw+"werner_v9"));
    return Array.from(new Uint8Array(buf)).map(b=>b.toString(16).padStart(2,"0")).join("");
  } catch {
    let h=5381; for(let i=0;i<pw.length;i++) h=((h<<5)+h)+pw.charCodeAt(i);
    return Math.abs(h).toString(16).padStart(16,"0");
  }
}

}

async function callAI({ messages, onChunk, signal, personaId, aiSettings, retryCount=0 }) {
  const persona = PERSONAS.find(p=>p.id===personaId)||PERSONAS[0];
  const settings = {...AI_DEFAULTS, ...getAISettings(), ...(aiSettings||{})};
  const recentTurns = Math.max(4, Math.min(20, Math.round(settings.memoryTurns||10)));

  // Construire l'historique de messages propre
  const history = messages
    .filter(m=>(m.role==="user"||m.role==="assistant")&&m.content&&String(m.content).trim())
    .slice(-recentTurns*2)
    .map(m=>({role:m.role==="assistant"?"assistant":"user", content:String(m.content||"").trim()}));

  // S'assurer que le premier message est "user"
  while(history.length>0 && history[0].role==="assistant") history.shift();
  if(history.length===0) return {raw:"", model:"Werner AI"};

  const systemPrompt = [
    SYSTEM_BASE,
    persona.prompt,
    getAiStyleInstruction(settings),
    settings.systemNote?.trim() ? `\nInstruction additionnelle : ${settings.systemNote.trim()}` : "",
  ].filter(Boolean).join("\n\n");

  try {
    if (signal?.aborted) throw new DOMException("Aborted","AbortError");

    // ✅ Appel API Anthropic natif (disponible dans les artifacts Claude)
    const res = await fetch(ANTHROPIC_URL, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        // Pas de clé API nécessaire dans l'environnement artifact Claude
      },
      signal,
      body: JSON.stringify({
        model: ANTHROPIC_MODEL,
        max_tokens: Math.round(settings.maxTokens)||2048,
        stream: false,
        system: systemPrompt,
        messages: history,
      }),
    });

    if (!res.ok) {
      const status = res.status;
      const errText = await res.text().catch(()=>"");
      
      // Retry sur surcharge
      if ([429, 529, 503, 502].includes(status)) {
        if (retryCount < 3) {
          const delay = Math.min(1000 * Math.pow(2, retryCount), 8000);
          await new Promise(r=>setTimeout(r, delay));
          return callAI({messages,onChunk,signal,personaId,aiSettings,retryCount:retryCount+1});
        }
        throw Object.assign(new Error("QUOTA"), {detail:"Le service est momentanément surchargé. Réessayez dans quelques instants."});
      }
      if (status === 401) throw Object.assign(new Error("AUTH"), {detail:"Erreur d'authentification. Veuillez recharger la page."});
      if (status === 400) throw Object.assign(new Error("INVALID"), {detail:"Requête invalide. Essayez de reformuler votre question."});
      throw Object.assign(new Error("API"), {detail:`Erreur serveur (${status}). Réessayez.`});
    }

    const data = await res.json();
    
    // Extraire le texte de la réponse
    let raw = "";
    if (data?.content && Array.isArray(data.content)) {
      raw = data.content
        .filter(b=>b.type==="text")
        .map(b=>b.text||"")
        .join("");
    } else if (typeof data?.content === "string") {
      raw = data.content;
    }

    if (!raw || !raw.trim()) {
      // Retry si réponse vide
      if (retryCount < 2) {
        await new Promise(r=>setTimeout(r, 500));
        return callAI({messages,onChunk,signal,personaId,aiSettings,retryCount:retryCount+1});
      }
      throw Object.assign(new Error("EMPTY"), {detail:"La réponse était vide. Réessayez."});
    }

    // Simuler un streaming fluide pour l'UX
    if (onChunk) {
      const chunks = raw.split(/(?<=[\s,.!?;:])/).filter(Boolean);
      let acc = "";
      for (const chunk of chunks) {
        if (signal?.aborted) break;
        acc += chunk;
        onChunk(acc);
        if (chunks.length > 10) {
          await new Promise(r=>setTimeout(r, Math.max(5, 15 - Math.floor(raw.length/200))));
        }
      }
      if (!signal?.aborted) onChunk(raw); // Assurer la réponse complète
    }

    return { raw, model: "Werner AI" };

  } catch(err) {
    if (err?.name === "AbortError") throw err;
    
    // Retry générique sur erreur réseau
    if (retryCount < 2 && !err.detail) {
      await new Promise(r=>setTimeout(r, 800*(retryCount+1)));
      return callAI({messages,onChunk,signal,personaId,aiSettings,retryCount:retryCount+1});
    }
    throw err;
  }
}

  return {thinking:match?match[1].trim():null, answer:raw.replace(/<w-think>[\s\S]*?<\/w-think>/gi,"").trim()};
}

// ── MARKDOWN RENDERER ─────────────────────────────────────────────────────────
const LANG_COLORS={javascript:"#f7df1e",js:"#f7df1e",typescript:"#3178c6",ts:"#3178c6",python:"#3776ab",py:"#3776ab",html:"#e34c26",css:"#264de4",jsx:"#61dafb",tsx:"#61dafb",react:"#61dafb",bash:"#4eaa25",sh:"#4eaa25",json:"#f59e0b",sql:"#336791",rust:"#ce422b",go:"#00acd7",java:"#b07219",cpp:"#f34b7d",c:"#555555",php:"#777bb4",ruby:"#cc342d",swift:"#f05138",kotlin:"#7f52ff"};

function renderMD(text) {
  if(!text) return "";
  return text
    .replace(/```(\w*)\n?([\s\S]*?)```/g,(_,lang,body)=>{
      const l=(lang||"code").toLowerCase(), color=LANG_COLORS[l]||"#8b5cf6";
      const esc=body.trim().replace(/</g,"&lt;").replace(/>/g,"&gt;");
      const enc=encodeURIComponent(body.trim()), lines=body.trim().split("\n").length;
      return `<div class="cb"><div class="cb-bar"><div class="cb-dots"><span></span><span></span><span></span></div><span class="cb-lang" style="color:${color}">${l}</span><span class="cb-lines">${lines}L</span><button class="cb-copy" onclick="this.textContent='✓';setTimeout(()=>this.textContent='copy',2e3);navigator.clipboard.writeText(decodeURIComponent('${enc}'))">copy</button></div><code>${esc}</code></div>`;
    })
    .replace(/`([^`]+)`/g,'<code class="ic">$1</code>')
    .replace(/\*\*\*([^*]+)\*\*\*/g,"<strong><em>$1</em></strong>")
    .replace(/\*\*([^*]+)\*\*/g,"<strong>$1</strong>")
    .replace(/\*([^*]+)\*/g,"<em>$1</em>")
    .replace(/^### (.+)$/gm,'<h3 class="md-h3">$1</h3>')
    .replace(/^## (.+)$/gm,'<h2 class="md-h2">$1</h2>')
    .replace(/^# (.+)$/gm,'<h1 class="md-h1">$1</h1>')
    .replace(/^\d+\. (.+)$/gm,'<li class="md-li ol">$1</li>')
    .replace(/^[-•*] (.+)$/gm,'<li class="md-li">$1</li>')
    .replace(/(<li[\s\S]*?<\/li>)/g,'<ul class="md-ul">$1</ul>')
    .replace(/^> (.+)$/gm,'<blockquote class="md-bq">$1</blockquote>')
    .replace(/---/g,'<hr class="md-hr"/>')
    .replace(/\n\n+/g,'<br/><br/>')
    .replace(/\n/g,'<br/>')
    .replace(/\[([^\]]+)\]\(([^)]+)\)/g,'<a href="$2" target="_blank" rel="noopener" class="md-a">$1</a>');
}

// ── CHAT PAGE ─────────────────────────────────────────────────────────────────
function ChatPage({user,setUser,convs,setConvs,activeConvId,setActiveConvId,showToast,onUpgrade,persona,promptToSend,clearPrompt}) {
  const [msgs,setMsgs]=useState([]);
  const [input,setInput]=useState("");
  const [loading,setLoading]=useState(false);
  const [streamTxt,setStreamTxt]=useState("");
  const [isThinking,setIsThinking]=useState(false);
  const [files,setFiles]=useState([]);
  const [recording,setRecording]=useState(false);
  const [tts,setTts]=useState(false);
  const [copied,setCopied]=useState(null);
  const [artifacts,setArtifacts]=useState([]);
  const [showArt,setShowArt]=useState(false);
  const [dragOver,setDragOver]=useState(false);
  const [retryMsg,setRetryMsg]=useState(null);
  const abortRef=useRef(null),endRef=useRef(null),taRef=useRef(null),fileRef=useRef(null),mrRef=useRef(null);

  useEffect(()=>{if(activeConvId)setMsgs(DB.get("w_conv_"+activeConvId,[]));else setMsgs([]);},[activeConvId]);
  useEffect(()=>{endRef.current?.scrollIntoView({behavior:"smooth"});},[msgs,streamTxt]);
  useEffect(()=>{
    if(promptToSend){
      setInput(promptToSend);
      clearPrompt?.();
      setTimeout(()=>taRef.current?.focus(),100);
    }
  },[promptToSend]);

  const saveConv=(id,m,title)=>{
    DB.set("w_conv_"+id,m);
    setConvs(prev=>{
      const ex=prev.find(c=>c.id===id);
      if(ex)return prev.map(c=>c.id===id?{...c,title:title||c.title}:c);
      return [{id,title:title||"Conversation",ts:Date.now()},...prev];
    });
  };

  const {allowed,count,limit,isPrem}=checkLimit(user);

  const send=async(txt=input,xfiles=files,isRetry=false)=>{
    const content=sanitizeInput((txt||"").trim());
    if(!content&&xfiles.length===0) return;
    if(loading) return;

    if(!allowed&&user.role!=="admin"){
      if(checkLimit(user).burstBlocked){showToast("⚡ Trop rapide — attendez quelques secondes.","error");return;}
      showToast(`Limite de ${limit} messages atteinte. Revenez demain ou passez Premium.`,"error");
      onUpgrade();return;
    }

    if(moderate(content)){
      setMsgs(m=>[...m,
        {id:Date.now()-1,role:"user",content},
        {id:Date.now(),role:"assistant",content:"⚠️ Cette demande ne peut pas être traitée.",blocked:true}
      ]);
      setInput("");setFiles([]);return;
    }

    let fTxt="";
    for(const f of xfiles) fTxt+=`\n\n[Fichier: ${f.name}]\n${f.content}`;
    const userContent=content+(fTxt?`\n\n---\n${fTxt}`:"");
    const userMsg={id:Date.now(),role:"user",content:userContent,files:xfiles.map(f=>f.name)};
    
    const baseMs = isRetry ? msgs.filter(m=>!m.isError) : msgs;
    const newMsgs=[...baseMs,userMsg];
    setMsgs(newMsgs);setInput("");setFiles([]);setRetryMsg(null);
    setLoading(true);setStreamTxt("");setIsThinking(true);

    incrementCount(user.id);
    RateLimit.record(`burst_${user.id}`);
    const updUser={...user,msgCount:(user.msgCount||0)+1};
    saveUser(updUser);setUser(updUser);DB.set("w_current_user",updUser);

    const convId=activeConvId||crypto.randomUUID();
    if(!activeConvId)setActiveConvId(convId);
    abortRef.current=new AbortController();
    const safetyTimer=setTimeout(()=>abortRef.current?.abort(),60000);

    try {
      const aiSettings=getAISettings();
      setIsThinking(false);

      const {raw}=await callAI({
        messages:newMsgs.map(m=>({role:m.role,content:m.content})),
        signal:abortRef.current.signal,
        onChunk:chunk=>{setStreamTxt(chunk);},
        personaId:persona,
        aiSettings,
      });
      clearTimeout(safetyTimer);

      const {thinking,answer}=parseResponse(raw);
      const arts=extractArtifacts(answer);
      const asstMsg={id:Date.now()+1,role:"assistant",content:answer,thinking,artifacts:arts,model:"Werner AI"};
      const finalMsgs=[...newMsgs,asstMsg];
      setMsgs(finalMsgs);setStreamTxt("");
      if(arts.length>0){setArtifacts(arts);setShowArt(true);}
      if(tts&&answer)speakText(answer);
      saveConv(convId,finalMsgs,content.slice(0,50)+(content.length>50?"…":""));

    } catch(err) {
      clearTimeout(safetyTimer);
      setStreamTxt("");setIsThinking(false);
      if(err.name==="AbortError"){setLoading(false);return;}
      
      // Message d'erreur clair et bouton retry
      const errDetail = err.detail || (
        err.message==="QUOTA" ? "Service surchargé. Réessayez dans quelques instants." :
        err.message==="AUTH"  ? "Erreur d'authentification. Rechargez la page." :
        err.message==="EMPTY" ? "Réponse vide reçue. Réessayez." :
        "Erreur réseau. Vérifiez votre connexion et réessayez."
      );
      const asstMsg={id:Date.now()+2,role:"assistant",isError:true,errorDetail:errDetail,retryData:{txt:content,files:xfiles}};
      setMsgs(m=>[...m,asstMsg]);
      showToast("Une erreur est survenue.","error");
    } finally{setLoading(false);setIsThinking(false);}
  };

  const retry=(retryData)=>{
    setMsgs(m=>m.filter(msg=>!msg.isError));
    send(retryData.txt,retryData.files||[],true);
  };

  const stopGen=()=>abortRef.current?.abort();

  const speakText=text=>{
    const clean=text.replace(/<[^>]+>/g,"").replace(/[#*`]/g,"").slice(0,3000);
    const utt=new SpeechSynthesisUtterance(clean);
    utt.lang="fr-FR";utt.rate=1.0;
    const voices=window.speechSynthesis.getVoices();
    const fr=voices.find(v=>v.lang.startsWith("fr"))||voices[0];
    if(fr)utt.voice=fr;
    window.speechSynthesis.cancel();window.speechSynthesis.speak(utt);
  };
  const stopTts=()=>window.speechSynthesis.cancel();

  const startRec=async()=>{
    try {
      const stream=await navigator.mediaDevices.getUserMedia({audio:true});
      const mr=new MediaRecorder(stream),chunks=[];
      mr.ondataavailable=e=>chunks.push(e.data);
      mr.onstop=()=>{stream.getTracks().forEach(t=>t.stop());showToast("🎤 Audio enregistré.");setInput(p=>p+" [Message vocal]");};
      mr.start();mrRef.current=mr;setRecording(true);
    } catch{showToast("Microphone non autorisé.","error");}
  };
  const stopRec=()=>{mrRef.current?.stop();setRecording(false);};

  const addFiles=async selected=>{
    const parsed=await Promise.all(Array.from(selected).map(async f=>{
      if(f.type.startsWith("image/"))return{name:f.name,content:"[Image jointe]",type:"image"};
      const t=await f.text().catch(()=>"[Contenu non lisible]");
      return{name:f.name,content:t.slice(0,12000),type:"text"};
    }));
    setFiles(p=>[...p,...parsed]);
  };

  const copyMsg=async(content,id)=>{await navigator.clipboard.writeText(content).catch(()=>{});setCopied(id);setTimeout(()=>setCopied(null),2000);};
  const autoSz=()=>{const t=taRef.current;if(t){t.style.height="22px";t.style.height=Math.min(t.scrollHeight,150)+"px";}};

  const limitPct=!isPrem&&user.role!=="admin"?Math.min(100,(count/limit)*100):0;
  const currentPersonaData=PERSONAS.find(p=>p.id===persona)||PERSONAS[0];
  const aiSettings=getAISettings();
  const aiStyleLabel=aiSettings.responseStyle==="creative"?"Créatif":aiSettings.responseStyle==="precise"?"Précis":"Équilibré";

  const SUGG=[
    {icon:"💡",t:"Brainstorming",s:"Idées créatives et innovation"},
    {icon:"💻",t:"Code expert",s:"Développement, debug, archi"},
    {icon:"📝",t:"Rédaction pro",s:"Articles, rapports, emails"},
    {icon:"🌐",t:"Analyse et stratégie",s:"Tendances, SWOT, décisions"},
  ];

  return (
    <div className="main">
      <div className="topbar">
        <div className="tb-title">{convs.find(c=>c.id===activeConvId)?.title||"Werner AI"}</div>
        <div style={{display:"flex",alignItems:"center",gap:5,padding:"3px 9px",background:"var(--surface)",borderRadius:20,fontSize:11,fontWeight:600,color:"var(--text2)",border:"1px solid var(--border)",flexShrink:0}}>
          <span>{currentPersonaData.icon}</span><span style={{color:"var(--text)"}}>{currentPersonaData.name}</span>
        </div>
        <div style={{display:"flex",alignItems:"center",gap:5,padding:"3px 9px",background:"var(--surface)",borderRadius:20,fontSize:11,fontWeight:600,color:"var(--text2)",border:"1px solid var(--border)",flexShrink:0}}>
          <span>🎛️</span><span style={{color:"var(--text)"}}>{aiStyleLabel}</span>
        </div>
        <div className="werner-badge"><div className="werner-dot"/>Werner AI</div>
        <button className="ib" onClick={()=>{setTts(!tts);if(tts)stopTts();}} title={tts?"Désactiver voix":"Activer voix"}>
          <Ic n={tts?"volume":"volume_x"} s={14} c={tts?"var(--a5)":"currentColor"}/>
        </button>
        {artifacts.length>0&&<button className="ib" onClick={()=>setShowArt(!showArt)}><Ic n="code" s={14} c={showArt?"var(--a)":"currentColor"}/></button>}
        {loading&&<button className="ib" onClick={stopGen} title="Arrêter"><Ic n="stop" s={14} c="#f87171"/></button>}
      </div>

      <div className="chat-wrap">
        <div className="chat-main">
          <div className="chat-body"
            onDragOver={e=>{e.preventDefault();setDragOver(true);}}
            onDragLeave={()=>setDragOver(false)}
            onDrop={e=>{e.preventDefault();setDragOver(false);if(e.dataTransfer.files.length)addFiles(e.dataTransfer.files);}}>

            {dragOver&&<div className="drag-ov"><div style={{fontSize:40}}>📎</div><div className="drag-txt">Déposez vos fichiers</div></div>}

            {msgs.length===0&&!loading?(
              <div className="welcome">
                <div className="welcome-logo"><WLogo s={72}/></div>
                <div className="welcome-title">Bonjour, {user.username} 👋</div>
                <div className="welcome-panel">

              {label:"Stockage",sub:`${DB.keys().length} entrées`,action:()=>{DB.keys().filter(k=>k!=="w_current_user").forEach(k=>DB.del(k));showToast("Stockage vidé.","success");},btnLabel:"Vider",cls:"btn-danger"},
              {label:"Conversations",sub:"Effacer l'historique",action:()=>{DB.keys().filter(k=>k.startsWith("w_conv_")).forEach(k=>DB.del(k));showToast("Conversations supprimées.","success");},btnLabel:"Effacer",cls:"btn-danger"},
            ].map((item,i)=>(
              <div key={i} style={{display:"flex",justifyContent:"space-between",alignItems:"center",padding:"10px 0",borderBottom:"1px solid var(--border)"}}>
                <div><div style={{fontSize:12.5,fontWeight:600,color:"var(--text)"}}>{item.label}</div><div style={{fontSize:11,color:"var(--text3)",marginTop:1}}>{item.sub}</div></div>
                {item.val||<button className={`btn btn-sm ${item.cls}`} onClick={item.action}>{item.btnLabel}</button>}
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}

// ── MAIN APP ──────────────────────────────────────────────────────────────────
export default function WernerApp() {
  const [user,setUser]=useState(()=>{
    const stored=DB.get("w_current_user",null);
    if(stored?._sessionExpiry&&Date.now()>stored._sessionExpiry){DB.del("w_current_user");return null;}
    return stored;
  });
  const [page,setPage]=useState("chat");
  const [convs,setConvs]=useState(()=>DB.get("w_convs",[]));
  const [activeConvId,setActiveConvId]=useState(null);
  const [sidebarOpen,setSidebarOpen]=useState(false);
  const [toasts,setToasts]=useState([]);
  const [showPricing,setShowPricing]=useState(false);
  const [showSettings,setShowSettings]=useState(false);
  const [persona,setPersona]=useState(()=>DB.get("w_persona","standard"));
  const [promptToSend,setPromptToSend]=useState(null);

  useEffect(()=>{DB.set("w_convs",convs);},[convs]);
  useEffect(()=>{DB.set("w_persona",persona);},[persona]);
  useEffect(()=>{
    const t=DB.get("w_theme","dark");
    if(t&&t!=="dark")document.documentElement.setAttribute("data-theme",t);
  },[]);

  const showToast=useCallback((msg,type="")=>{
    const id=Date.now();
    setToasts(t=>[...t,{id,msg,type}]);
    setTimeout(()=>setToasts(t=>t.filter(x=>x.id!==id)),3800);
  },[]);

  const handleLogin=u=>{const s={...u,_sessionExpiry:Date.now()+SESSION_TTL};DB.set("w_current_user",s);setUser(s);};
  const handleLogout=()=>{DB.del("w_current_user");setUser(null);setPage("chat");setActiveConvId(null);};
  const newChat=()=>{setActiveConvId(null);setPage("chat");};
  const delConv=id=>{setConvs(c=>c.filter(x=>x.id!==id));DB.del("w_conv_"+id);if(activeConvId===id)setActiveConvId(null);};
  const handleUpgrade=()=>{const updated={...user,isPremium:true};saveUser(updated);setUser(updated);DB.set("w_current_user",updated);setShowPricing(false);showToast("⚡ Werner Premium activé !","success");};
  const setUserPersist=u=>{setUser(u);DB.set("w_current_user",u);};

  const isPrem=user?.isPremium||user?.role==="admin";
  const count=user?getDailyCount(user.id):0;
  const limit=isPrem?LIMITS.premium:LIMITS.free;

  if(!user) return <><style>{STYLES}</style><AuthScreen onLogin={handleLogin}/></>;
  if(showPricing) return <><style>{STYLES}</style><PricingPage user={user} onUpgrade={handleUpgrade} onBack={()=>setShowPricing(false)}/></>;

  return (
    <>
      <style>{STYLES}</style>
      <div className="app">
        <Sidebar
          user={user} page={page} setPage={setPage}
          convs={convs} activeConvId={activeConvId}
          setActiveConvId={id=>{setActiveConvId(id);setPage("chat");}}
          onNew={newChat} onDelConv={delConv}
          open={sidebarOpen} setOpen={setSidebarOpen}
          onLogout={handleLogout} onPricing={()=>setShowPricing(true)}
          onSettings={()=>setShowSettings(true)}
          persona={persona} setPersona={setPersona}
          onSendPrompt={txt=>{setPromptToSend(txt);setPage("chat");setActiveConvId(null);}}
          count={count} limit={limit}
        />
        <div style={{flex:1,display:"flex",flexDirection:"column",overflow:"hidden",minWidth:0}}>
          {/* Mobile top bar */}
          <div className="mob-bar" style={{display:"none",padding:"9px 13px",borderBottom:"1px solid var(--border)",background:"rgba(0,0,0,.8)",alignItems:"center",gap:9,flexShrink:0,zIndex:10,backdropFilter:"blur(16px)"}}>
            <style>{`@media(max-width:768px){.mob-bar{display:flex!important;}}`}</style>
            <button className="ib" onClick={()=>setSidebarOpen(true)}><Ic n="menu" s={17}/></button>
            <div style={{flex:1,display:"flex",alignItems:"center",gap:7}}>
              <WLogo s={19}/><span style={{fontWeight:900,fontSize:13.5,background:"linear-gradient(120deg,#a78bfa,#818cf8,#38bdf8)",WebkitBackgroundClip:"text",WebkitTextFillColor:"transparent"}}>Werner AI</span>
            </div>
            <button className="ib" onClick={()=>setShowSettings(true)}><Ic n="settings" s={15}/></button>
            <button className="ib" onClick={handleLogout}><Ic n="logout" s={15}/></button>
          </div>

          {page==="chat"     &&<ChatPage     user={user} setUser={setUserPersist} convs={convs} setConvs={setConvs} activeConvId={activeConvId} setActiveConvId={setActiveConvId} showToast={showToast} onUpgrade={()=>setShowPricing(true)} persona={persona} promptToSend={promptToSend} clearPrompt={()=>setPromptToSend(null)}/>}
          {page==="profile"  &&<ProfilePage  user={user} setUser={setUserPersist} showToast={showToast} onUpgrade={()=>setShowPricing(true)}/>}
          {page==="projects" &&<ProjectsPage user={user} setPage={setPage} onNew={newChat} showToast={showToast}/>}
          {page==="guide"    &&<GuidePage/>}
          {page==="admin"    &&user.role==="admin"&&<AdminPage showToast={showToast}/>}
        </div>

        {showSettings&&<SettingsModal onClose={()=>setShowSettings(false)} user={user} showToast={showToast}/>}
        <Toasts toasts={toasts}/>
      </div>
    </>
  );
}
